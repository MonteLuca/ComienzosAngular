CurrencyPipe

ventasNetas | currency
ventasNetas | currency: 'ARG'
ventasNetas | currency: 'CAD' : 'symbol-narrow' : '1.4-4'
{{ventasNetas | currency}}
{{ventasNetas | currency:'ARG'}}
{{ventasNetas | currency:'CAD':'symbol-narrow':'1.4-4'}}
